package org.example;
import com.coupang.openapi.sdk.Hmac;
import org.apache.http.HttpEntity;
import org.apache.http.client.methods.CloseableHttpResponse;
import org.apache.http.client.methods.HttpGet;
import org.apache.http.client.utils.URIBuilder;
import org.apache.http.conn.scheme.Scheme;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.util.EntityUtils;
import org.springframework.web.util.UriComponents;
import org.springframework.web.util.UriComponentsBuilder;
import java.io.IOException;

class HmacOrderbyday {

    private static final String HOST = "api-gateway.coupang.com";
    private static final int PORT = 443;
    private static final String SCHEMA = "https";
    //replace with your own accessKey
    private static final String ACCESS_KEY = "your-access-key";
    //replace with your own secretKey
    private static final String SECRET_KEY = "your-secret-key";

    public void test() {
        //params
        String method = "GET";
        //replace with your own vendorId
        String path = "/v2/providers/openapi/apis/api/v5/vendors/{vendorId}/ordersheets";
        CloseableHttpClient client = null;
        try {
            //create client
            client = HttpClients.createDefault();
            UriComponentsBuilder builder = UriComponentsBuilder.newInstance()
                    .path(path)
                    .queryParam("createdAtFrom", "2025-12-09%2B09:00")
                    .queryParam("createdAtTo", "2025-12-19%2B09:00")
                    .queryParam("status", "INSTRUCT");

            /********************************************************/
            //authorize, demonstrate how to generate hmac signature here
            String authorization = Hmac.generate(method,builder.build().toUriString(),SECRET_KEY, ACCESS_KEY);
            //print out the hmac key
            System.out.println(authorization);
            /********************************************************/
            UriComponents finalUri = builder
                    .scheme(SCHEMA)
                    .host(HOST)
                    .port(PORT)
                    .build();
            HttpGet get = new HttpGet(finalUri.toUriString());
            System.out.println(finalUri.toUriString());
            /********************************************************/
            // set header, demonstarte how to use hmac signature here
            get.addHeader("Authorization", authorization);
            get.addHeader("X-Requested-By", vendorId);
            get.addHeader("X-MARKET", "KR");
            get.addHeader("content-type", "application/json");
            //add X-EXTENDED-TIMEOUT value
            get.addHeader("X-EXTENDED-TIMEOUT", "90000");
            System.out.println(get.toString());
            CloseableHttpResponse response = null;
            try {
                //execute get request
                response = client.execute(get);
                //print result
                System.out.println("status code:" + response.getStatusLine().getStatusCode());
                System.out.println("status message:" + response.getStatusLine().getReasonPhrase());
                HttpEntity entity = response.getEntity();
                System.out.println("result:" + EntityUtils.toString(entity));
            } catch (Exception e) {
                e.printStackTrace();
            } finally {
                if (response != null) {
                    try {
                        response.close();
                    } catch (IOException e) {
                        e.printStackTrace();
                    }
                }
            }
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            if (client != null) {
                try {
                    client.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
        }
    }

    public static void main(String[] args) {
        HmacOrderbyday hmacTest = new HmacOrderbyday();
        hmacTest.test();
    }
}
